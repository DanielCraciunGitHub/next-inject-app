export async function ApiKeyCommand() {
  return (
    <pre>
      <code>
        {`next-inject auth -k ${(await api.pluginRouter.getApiKey()).toString()}`}
      </code>
    </pre>
  )
}

export async function PluginCommands() {
  const plugins = await api.pluginRouter.getOwnedPlugins()

return plugins.map(function (plugin) {
return (

<>

<div key={plugin.originalName}>
  <Link href={`/plugins/${plugin.parsedName}`}>
    <Heading as="h3">{plugin.originalName}</Heading>
  </Link>
  <pre>
    <code>{`next-inject add ${plugin.parsedName}`}</code>
  </pre>
  <hr />
</div>

</>
)}) }
